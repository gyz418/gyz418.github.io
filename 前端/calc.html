      if (count2) {

var twoSum = function(nums, target) {
    const map = new Map();
    for(let i = 0, len = nums.length;i < len;i++) {
    // 一层遍历，用 target 减去每一项，在 map 的 key 中寻找
        if(map.has(target - nums[i])) {
        // 存在则返回结果
            console.log(11)
            return [map.get(target - nums[i]), i];
        }
        // 不存在，则设置 map key 和 val
        
        map.set(nums[i], i);
        console.log(22,map)
    }
    return [];
};


const twoSum2 = (nums, target) => {
  const len = nums.length
  for (let i = 0; i < len; i++) {
    // 因为同一元素不允许重复出现，所以从i的下一位开始遍历
    for (let j = i + 1; j < len ; j++) {
      // find the answer, return
      if (nums[ i] + nums[ j ] === target) {
        return [ i, j ]
      }
    }
  }
}

# 整数转罗马数字

贪心算法： 找最大值优先

I             1
V             5
X             10
L             50
C             100
D             500
M             1000


var intToRoman = function(num) {
    var res = ''
    var map = {
        1000: 'M',
        900: 'CM',
        500: 'D',
        400: 'CD',
        100: 'C',
        90: 'XC',
        50: 'L',
        40: 'XL',
        10: 'X',
        9: 'IX',
        5: 'V',
        4: 'IV',
        1: 'I',
    }
    var keys = Object.keys(map).reverse()

    for(let i = 0; i < keys.length; i++){
        while (num >= keys[i]) {
            res += map[keys[i]]
            num -= keys[i]   // 一直减去最大值
        }
    }

    return res
};

# 找钱贪心算法

let money = 132
function test(money){
  let arr = [1,2,5,10,20,50,100]  // 有的人民币
  let res = []
  arr = arr.reverse()  // 从大到小排序
  for(let i=0;i<arr.length;i++){
    while(money>=arr[i]){
      res.push(arr[i])   // 找到的零钱
      money-=arr[i]  // 一直减去最大值
    }
  }
  return res
}

# 最大子序和

输入:[-2,1,-3,4,-1,2,1,-5,4] 输出: 6 解释: 连续子数组 [4,-1,2,1] 的和最大，为 6。

var maxSubArray = function(nums) {
    var temp = nums[0]
    var sum = 0
    for(let i = 0; i < nums.length; i++){
        if(sum > 0){
            sum += nums[i] //组合>0,则是优势组合，保留
        }else{
            sum = nums[i] // 组合<0,则是劣势组合，直接覆盖
        }
        temp = Math.max(temp, sum),//维护最大的组合和或最大的单个值
    }
    return temp
};

/**
 * 取出右上角，如果大于目标值，这一列全排除
 * 如果小于目标值，上一行全排除
 */
function test(arr,res){
  // 取出右上角的值
  let arrLine = arr[0]
  let len = arrLine.length
  let num = arrLine[len-1]
  while(num>res){
    arr.forEach(val=>{
      val.pop()
    })
    num = arrLine[arrLine.length-1]  //  要有条件变动，不然，永远while(num>res)
  }
  // 小值
  while(num<res){
    arr.shift()
    arrLine = arr[0]
    if(!arrLine) return false    // 找不到结束
    num = arrLine[arrLine.length-1]
  }
  if(num===res){
    console.log('find');
    return arr
  }
  console.log('no')
  return arr
}

# js实现链表

// 链表
function Node(element){
  this.element = element;
  this.next = null
}

//链表类
function LList () {
  this.head = new Node( 'head' );     //头节点
  this.find = find;                   //查找节点
  this.insert = insert;               //插入节点
  // this.remove = remove;               //删除节点
  // this.findPrev = findPrev;           //查找前一个节点
  this.display = display;             //显示链表
}
function find ( item ) {
  var currNode = this.head;
  while ( currNode.element != item ){
    currNode = currNode.next;
  }
  return currNode;
}
function insert ( newElement , item ) {
  var newNode = new Node( newElement );
  var currNode = this.find( item );
  newNode.next = currNode.next;  // 注意先后顺序
  currNode.next = newNode;  // 注意先后顺序
}
function display () {
  var currNode = this.head;
  while ( !(currNode.next == null) ){
    console.log( currNode.next.element );   // 不打印第一个head元素  currNode.element 打印当前元素
    currNode = currNode.next;
  }
}
var fruits = new LList();

fruits.insert('Apple' , 'head');
fruits.insert('Banana' , 'Apple');
fruits.insert('Pear' , 'Banana');
fruits.insert('Cc' , 'Apple');

console.log(fruits.display());

// head next-> apple next -> banana next -> pear next
// head next-> apple next ->Cc next-> banana next -> pear next
