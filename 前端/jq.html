# append

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
<div id="a"></div>
<script src="https://cdn.bootcdn.net/ajax/libs/jquery/3.6.0/jquery.min.js"></script>
<script>
  let arr = [
    {name: 'kang', age: 12, list: ['aa', 'bb', 'cc']},
    {name: 'kang2', age: 13, list: ['aa', 'bb', 'ddd']}
  ]
  for (let i = 0; i < arr.length; i++) {
    $("#a").append(`
    <div>
        <p>${arr[i].name}</p>
        <p>${arr[i].age}</p>
        <div>${getList(arr[i].list)}</div>
    </div>
`)
  }
  function getList(list){
    let str = ''
    for(let i=0;i<list.length;i++){
      str+= `<p>${list[i]}</p>`
    }
    return str
  }
</script>
</body>
</html>
