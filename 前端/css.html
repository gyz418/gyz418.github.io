    // 旋转圆1 顺时针  动画时间不能一致，一个10s,一个6s
.round1 {
  width: 407px;
  height: 401px;
  background-image: url('../img/card/round1.png');
  margin: 47px auto 0;
  animation: 10s rotate1 linear infinite both;
  position: relative;
  z-index: 1;
  @keyframes rotate1 {
    from {
      transform: rotate(0deg);
    }
    50% {
      transform: rotate(180deg);
    }
    to {
      transform: rotate(360deg);
    }
  }
}
// 旋转圆2 逆时针
.round2 {
  position: absolute;
  top: 0;
  left: 0;
  right: 0;
  bottom: 0;

  width: 402px;
  height: 385px;
  background-image: url('../img/card/round2.png');

    from {
      transform: rotate(0deg);
    }

      transform: rotate(-180deg);
    }
    to {
      transform: rotate(-360deg);
    }
  }
}
