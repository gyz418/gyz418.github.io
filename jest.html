        })
      }
    }
  };

  })

// 条件判断
function getNum(num:any) {
  if (num) {
    return 10
  } else {
    return 5
  }
}
describe('getNum',()=>{
  test('getNum',()=>{
    const num = ''
    expect(getNum(num)).toBe(5)
    const param = 1
    expect(getNum(param)).toBe(10)
  })
})
// 本地存储
function getLocal(key: string) {
  return localStorage.getItem(key)
}
describe('getLocal', () => {
  test('getLocal', () => {
    // 自定义链接
    ;(global.window as typeof window) = Object.create(window)
    Object.defineProperty(window, 'localStorage', {
      value: {
        getItem: (key: string) => {
          if (key === 'abc') return 666
          if (key === 'ddd') return 888
        },
      },
    })
    expect(getLocal('abc')).toBe(666)
    expect(getLocal('ddd')).toBe(888)
  })
})

# setTimeout

// a.ts
let timer
export function getVal() {
  if (timer) {
    clearTimeout(timer)
    timer = null
  }
  timer = setTimeout(() => {
    return 10
  }, 1000)
}
// a.spec.ts
jest.useFakeTimers()
jest.spyOn(global, 'setTimeout')
test('getVal', () => {
  const { getVal } = require('./a')
  getVal()
  expect(setTimeout).toHaveBeenCalledTimes(1)
  jest.runAllTimers()
  getVal()
  expect(setTimeout).toHaveBeenCalledTimes(2)
})
