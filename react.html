# useState useEffect

React 16.8 替代class

import React, { useState, useEffect } from 'react';
function Example() {
  // 声明一个新的叫做 “count” 的 state 变量
  const [count, setCount] = useState(0);
  useEffect(()=>{
      console.log(666)  // 当useState数据变化，这里会再执行，跟vue3不一样，vue3的 effect()必须有响应式数据才会再次执行
  })
  return (
    <div>
      <p>You clicked {count} times</p>
      <button onClick={() => setCount(count + 1)}>
        Click me
      </button>
    </div>
  );
}

# useEffect

用useEffect来替换 componentDidMount() componentDidUpdate() componentWillUnmount()

可以返回一个函数，清除定时器等

useEffect(() => {
  document.title = `You clicked ${count} times`;
}, [count])  // 当count更改时才执行，这样就跟vue3一样了
// 传个[]空数组，也可以保证useEffect只执行一次

# reducers dispatch

页面

import {connect} from "dva";

const Test = ({stateNum,dispatch}) => {
  const add=()=>{
    dispatch({
      type:'test1/addNum',
    })
  }
  return (
    <div>
      <h1>num:{stateNum}</h1>
      <button onClick={add}>add</button>
    </div>
  )
}
export default connect(({test1}) => {
  return {
    stateNum: test1.num
  }
})(Test)

数据

export default {
  namespace: 'test1',
  state: {
    num:666
  },
  reducers:{
    addNum(state){
      state.num++
      return {
        ...state   // 当state是对象时，必须返回 {...state} 直接返回 state，页面不更新
      }
    },
  }
};

# effects

页面

import {connect} from "dva";

const Test = ({stateNum,dispatch}) => {
  const add=()=>{
    dispatch({
      type:'test1/getNum',  //调用effects
      payload:1000
    })
  }
  return (
    <div>
      <h1>num:{stateNum}</h1>
      <button onClick={add}>add</button>
    </div>
  )
}
export default connect(({test1}) => {
  return {
    stateNum: test1.num
  }
})(Test)

数据

export default {
  namespace: 'test1',
  state: {
    num:666
  },
  reducers:{
    addNum(state){
      state.num++
      return {
        ...state
      }
    },
    upDateNum(state,{payload}){
      state.num = payload
      return {
        ...state
      }
    }
  },
  effects:{
    * getNum({payload},{call,put,select}){
      const oldNum = yield select(state=>state.test1.num)   // 注意命名空间
      console.log(oldNum);
      yield put({     // put：发出一个 Action，类似于 dispatch
        type:'upDateNum',  // 本文件不需要命名空间
        payload:oldNum+payload  
      })
    }
  }
};
