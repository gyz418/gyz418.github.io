# demo

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
  <style>
    body{margin:0;}
  </style>
</head>
<script src="https://threejs.org/build/three.js"></script>
<body>
<script>
  var scene = new THREE.Scene();  // 场景
  var camera = new THREE.PerspectiveCamera( 75, window.innerWidth/window.innerHeight, 0.1, 1000 );  // 相机

  var renderer = new THREE.WebGLRenderer();   // 渲染器
  renderer.setSize( window.innerWidth, window.innerHeight );  // 渲染器尺寸
  document.body.appendChild( renderer.domElement );  // 添加到body

  var geometry = new THREE.BoxGeometry( 1, 1, 1 );   // 立方体
  var material = new THREE.MeshBasicMaterial( { color: 0x00ff00 } ); // 材质，立方体加颜色
  var cube = new THREE.Mesh( geometry, material );  // 网格，整合立方体和材质
  scene.add( cube );  // 场景添加加网格

  camera.position.z = 5;    // 相机位置 

  var animate = function () {
    requestAnimationFrame( animate );

    cube.rotation.x += 0.01;    // 网格旋转
    cube.rotation.y += 0.01;

    renderer.render( scene, camera );    // 渲染器渲染场景和相机
  };

  animate();
</script>
</body>
</html>

把立方体移到div中

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
  <style>
    body{margin:0;}
    .a{
        border: 1px solid #f00;
        width:400px;
        height: 400px;
        box-sizing: border-box;
    }
  </style>
</head>
<script src="https://threejs.org/build/three.js"></script>
<body>
<div class="a" id="a"></div>
<script>
  window.onload=function(){
    showBox()
  }
  function showBox(){
    var a = document.getElementById('a')
    var scene = new THREE.Scene();
    var camera = new THREE.PerspectiveCamera( 75, a.offsetWidth/a.offsetHeight, 0.1, 1000 );

    var renderer = new THREE.WebGLRenderer();
    renderer.setSize( a.offsetWidth, a.offsetHeight );
    a.appendChild(renderer.domElement)  // 加在a里面
    var geometry = new THREE.BoxGeometry( 1, 1, 1 );
    var material = new THREE.MeshBasicMaterial( { color: 0x00ff00 } );
    var cube = new THREE.Mesh( geometry, material );
    scene.add( cube );

    camera.position.z = 5;
    cube.rotateX(45)
    cube.rotateY(45)
    renderer.render( scene, camera );
  }
</script>
</body>
</html>

# demo02

obj加载，换相机

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <style>
        body {
            margin: 0;
        }

        .a {
            border: 1px solid #f00;
            width: 400px;
            height: 400px;
            box-sizing: border-box;
        }
    </style>
</head>
<script src="https://threejs.org/build/three.js"></script>
<script src="http://www.yanhuangxueyuan.com/threejs/examples/js/loaders/OBJLoader.js"></script>
<script src="http://www.yanhuangxueyuan.com/threejs/examples/js/loaders/MTLLoader.js"></script>
<body>
<div class="a" id="a"></div>
<script>

  window.onload = function () {
    showBox()
  }

  function showBox() {
    var a = document.getElementById('a')
    var scene = new THREE.Scene();

    // obj模型
    var loader = new THREE.OBJLoader()

    var width = a.offsetWidth; //窗口宽度
    var height = a.offsetHeight; //窗口高度
    var k = width / height; //窗口宽高比
    var s = 100; //三维场景显示范围控制系数，系数越大，显示的范围越大
    //创建相机对象
    var camera = new THREE.OrthographicCamera(-s * k, s * k, s, -s, 1, 1000);
    camera.position.set(100, 100, 100); //设置相机位置
    camera.lookAt(scene.position); //设置相机方向(指向的场景对象)

    // 渲染器
    var renderer = new THREE.WebGLRenderer();
    renderer.setSize(width, height); //设置渲染区域尺寸
    renderer.setClearColor(0xccbaaa, 1); //设置背景颜色
    a.appendChild(renderer.domElement); //body元素中插入canvas对象

    // 显示
    loader.load('./file.obj', (res) => {
      scene.add(res)
      renderer.render(scene,camera)
    })
  }
</script>
</body>
</html>

# demo03

加光变白色，加材质补充颜色等

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <style>
        body {
            margin: 0;
        }

        .a {
            border: 1px solid #f00;
            width: 400px;
            height: 400px;
            box-sizing: border-box;
        }
    </style>
</head>
<script src="https://threejs.org/build/three.js"></script>
<script src="http://www.yanhuangxueyuan.com/threejs/examples/js/loaders/OBJLoader.js"></script>
<script src="http://www.yanhuangxueyuan.com/threejs/examples/js/loaders/MTLLoader.js"></script>
<body>
<div class="a" id="a"></div>
<script>

  window.onload = function () {
    showBox()
  }

  function showBox() {
    var a = document.getElementById('a')
    var scene = new THREE.Scene();

    // obj模型
    var loader = new THREE.OBJLoader()

    var width = a.offsetWidth; //窗口宽度
    var height = a.offsetHeight; //窗口高度
    var k = width / height; //窗口宽高比
    var s = 100; //三维场景显示范围控制系数，系数越大，显示的范围越大
    //创建相机对象
    var camera = new THREE.OrthographicCamera(-s * k, s * k, s, -s, 1, 1000);
    camera.position.set(100, 100, 100); //设置相机位置
    camera.lookAt(scene.position); //设置相机方向(指向的场景对象)

    // 渲染器
    var renderer = new THREE.WebGLRenderer();
    renderer.setSize(width, height); //设置渲染区域尺寸
    renderer.setClearColor(0xccbaaa, 1); //设置背景颜色（背景默认黑色，obj加载出来也是黑色）
    a.appendChild(renderer.domElement); //body元素中插入canvas对象

    // 【先加材料(颜色等)再加obj】
    var mtlLoader = new THREE.MTLLoader()
    mtlLoader.setCrossOrigin('Anonymous')
    mtlLoader.load('./file.mtl', (materials) => {
      materials.preload()
      loader.setMaterials(materials)
      loadObj()
    })

    function loadObj() {
      loader.load('./file.obj', (res) => {
        scene.add(res)  // 默认是黑色
        //环境光
        var ambient = new THREE.AmbientLight(0x777777);
        scene.add(ambient);

        var directionalLight = new THREE.DirectionalLight(0xffffff, 1);
        // 设置光源的方向：通过光源position属性和目标指向对象的position属性计算
        directionalLight.position.set(200, 100, 200);
        directionalLight.target = res;
        scene.add(directionalLight);

        // 加光变白色
        renderer.render(scene, camera)
      })
    }
  }
</script>
</body>
</html>
